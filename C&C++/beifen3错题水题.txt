#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i=0,j=0,k=0,l=0;
    int n;
    while((n=getchar())!='\n'){       //哈哈哈哈神他妈的反斜杠 是'\'啊'\' 是键盘上比较少的那个啊啊哈哈哈哈     （斜杠是撇 反斜杠是捺）
            if((n>='a'&&n<='z') || (n>='A'&&n<='B'))
            i++;
            else if(n>='0'&&n<='9')
                j++;
            else if(n==' ')
                    k++;
            else
                        l++;
    }
    cout<<i<<" "<<j<<" "<<k<<" "<<l;
    return 0;
}
